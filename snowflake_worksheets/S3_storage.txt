// 1.Create integration object of the right STORAGE_PROVIDER
// 2. DESC the integration object to setup access/rights on the provider side
// 3. Create and use the STAGE using the integration object

// Create storage integration object

create or replace storage integration s3_int
  TYPE = EXTERNAL_STAGE
  STORAGE_PROVIDER = S3
  ENABLED = TRUE 
  STORAGE_AWS_ROLE_ARN = 'arn:aws:iam::224849216410:role/snowflake-access' //fill this from AWS IAM r√¥le created
  STORAGE_ALLOWED_LOCATIONS = ('s3://snowflake-udemy-vmzl/csv/', 's3://snowflake-udemy-vmzl/json/')
   COMMENT = 'This an optional comment' 
   
   
// See storage integration properties to fetch external_id so we can update it in S3
DESC integration s3_int;


//Load data from AWS S3
//
//
// Create table first
CREATE OR REPLACE TABLE movie_titles (
  show_id STRING,
  type STRING,
  title STRING,
  director STRING,
  cast STRING,
  country STRING,
  date_added STRING,
  release_year STRING,
  rating STRING,
  duration STRING,
  listed_in STRING,
  description STRING );
  
  
// Create file format object
CREATE OR REPLACE file format csv_fileformat
    type = csv
    field_delimiter = ','
    skip_header = 1
    null_if = ('NULL','null')
    empty_field_as_null = TRUE    
    FIELD_OPTIONALLY_ENCLOSED_BY = '"';
    
    
 // Create stage object with integration object & file format object
CREATE OR REPLACE stage csv_folder
    URL = 's3://snowflake-udemy-vmzl/csv/'
    STORAGE_INTEGRATION = s3_int
    FILE_FORMAT = csv_fileformat;

// Use Copy command       
COPY INTO movie_titles
    FROM @csv_folder;
    

    
    
SELECT * FROM movie_titles
